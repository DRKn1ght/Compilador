int fibonacci_helper(int n, int a, int b) {
    if (n != 0) {
        print(a);
        print(" - ");
        fibonacci_helper(n-1, b, a+b);
    }
    return 0;
}

int fibonacci(int n) {
    fibonacci_helper(n, 0, 1);
    print("\n");
    return 0;
}

int start() {
    int n = 10;
    fibonacci(n);
    return 0;
}
